  • ✅ Controller KHÔNG gọi DAO trực tiếp — mọi DB access đi qua API

② ApiClient — Background Thread

  • 5

  • 6
    Tạo HttpRequest POST URI: http://localhost:8080/auction/bid, Content-Type: application/json
  • 7

③ Server Handler — Server Thread Pool

  • 9
    AuctionHandler.handle() nhận request path.endsWith("/auction/bid") && "POST" → gọi handleBid()
  • 10
    Parse body JSON JsonUtil.parseObject(readBody(ex)) → lấy auctionId, bidderId, amount
  • 11
    Lấy Auction từ AUCTION_CACHE getCachedAuction(auctionId) — nếu chưa có thì load từ DB, cache lại

    Gọi BiddingService.placeBid(auction, bidder, amount) Delegate toàn bộ business logic sang Service

④ Service + DAO + DB

  • 13
    BiddingService: acquire ReentrantLock auction.getLock().lock() — chỉ 1 thread xử lý bid cho auction này tại 1 thời điểm
  • 14
    Validate business rules Auction OPEN/RUNNING? Seller không tự bid? amount > currentPrice? endTime chưa qua?
  • 15
    BEGIN TRANSACTION conn.setAutoCommit(false) — bắt đầu transaction
  • 16
    AuctionDAO: UPDATE auction SET current_price=1500, highest_bidder_id=7 WHERE id=42
  • 17
    BidTransactionDAO: INSERT bid record INSERT INTO bid_transaction (auction_id, bidder_id, amount, bid_time)
  • 18
    AdvancedAuctionService: processAutoBids() Xử lý auto-bid nếu có; applyAntiSniping nếu < 10s còn lại
  • 19
    COMMIT + release lock conn.commit() → finally: auction.getLock().unlock()
Response ngược về Controller
  • 20
    Handler: gửi response JSON sendJson(ex, 200, {"success":true, "newPrice":1500.0})
  • 21
    ApiClient: parse response resp.get("newPrice").getAsDouble() → 1500.0 → trả về cho Task
  • 22
    Controller: Platform.runLater() update UI priceLabel.setText("1,500.00 USD"), enable button, show success message

① Controller — Gọi API trong Task
// AuctionDetailController.java
private void handlePlaceBid() {
    final double finalBid = bid;
    placeBidButton.setDisable(true);

    // Task chạy trên BACKGROUND thread — không block UI
    Task<Double> task = new Task<>() {
        @Override protected Double call() throws Exception {
            JsonObject resp = ApiClient.post("/auction/bid", Map.of(
                "auctionId", currentAuction.getId(),
                "bidderId",  SessionManager.getInstance().getUserId(),
                "amount",    finalBid));
            if (!resp.get("success").getAsBoolean())
                throw new RuntimeException(resp.get("error").getAsString());
            return resp.get("newPrice").getAsDouble();
        }
    };
    task.setOnSucceeded(e -> Platform.runLater(() -> {
        // Chạy lại trên JavaFX Application Thread — safe để update UI
        double newPrice = task.getValue();
        currentAuction.setCurrentPrice(newPrice);
        priceLabel.setText(String.format("%,.2f USD", newPrice));
        placeBidButton.setDisable(false);
    }));
    task.setOnFailed(e -> Platform.runLater(() -> {
        bidValidationLabel.setText(task.getException().getMessage());
        placeBidButton.setDisable(false);
    }));
    new Thread(task).start();
}
② ApiClient — HTTP Layer
// ApiClient.java
public static JsonObject post(String path, Map<String, Object> body) throws Exception {
    // Serialize body thành JSON
    String bodyJson = GSON.toJson(body);  // {"auctionId":42,"bidderId":7,"amount":1500.0}

    HttpRequest request = HttpRequest.newBuilder()
        .uri(URI.create(serverUrl + path))   // http://localhost:8080/auction/bid
        .header("Content-Type", "application/json")
        .timeout(Duration.ofSeconds(15))
        .POST(HttpRequest.BodyPublishers.ofString(bodyJson))
        .build();

    HttpResponse<String> response = HTTP.send(request,
        HttpResponse.BodyHandlers.ofString());

    return GSON.fromJson(response.body(), JsonObject.class);
    // {"success":true,"newPrice":1500.0}
}
③ Server Handler — Route + Delegate
// AuctionHandler.java
private void handleBid(HttpExchange ex) throws IOException {
    // Parse JSON body
    JsonObject body  = JsonUtil.parseObject(readBody(ex));
    int    auctionId = body.get("auctionId").getAsInt();
    int    bidderId  = body.get("bidderId").getAsInt();
    double amount   = body.get("amount").getAsDouble();

    // Lấy Auction từ cache (giữ lock chung giữa các request)
    Auction auction = getCachedAuction(auctionId);

    // Delegate sang Service — Handler KHÔNG xử lý business logic
    Bidder bidder = new Bidder(); bidder.setId(bidderId);
    new BiddingService().placeBid(auction, bidder, amount);

    // Build + gửi response
    Map<String, Object> resp = ok();
    resp.put("newPrice", auction.getCurrentPrice());
    sendJson(ex, 200, resp);
}
④ BiddingService — Business Logic + Lock + Transaction
// BiddingService.java
public void placeBid(Auction auction, Bidder bidder, double amount) {
    auction.getLock().lock();  // ← chỉ 1 thread vào được tại một thời điểm
    try (Connection conn = DatabaseConnection.getConnection()) {
        conn.setAutoCommit(false);  // BEGIN TRANSACTION
        try {
            // ── Validate business rules ──────────────────────────────────
            if (auction.getSeller().getId() == bidder.getId())
                throw new SellerBiddingOwnItemException(...);
            if (!auction.isOpen())
                throw new AuctionClosedException(...);
            if (amount <= auction.getCurrentPrice())
                throw new InvalidBidException(...);

            // ── Cập nhật Auction object trong memory ─────────────────────
            auction.setCurrentPrice(amount);
            auction.setHighestBidder(bidder);

            // ── DAO: UPDATE auction trong DB ──────────────────────────────
            new AuctionDAO().updateBid(conn, auction.getId(), amount, bidder.getId());

            // ── DAO: INSERT bid_transaction ───────────────────────────────
            BidTransaction bt = new BidTransaction(bidder, auction, amount, LocalDateTime.now());
            new BidTransactionDAO().insert(bt, conn);

            // ── Auto-bid + Anti-sniping ───────────────────────────────────
            new AdvancedAuctionService().processAutoBids(conn, auction);
            new AdvancedAuctionService().applyAntiSniping(conn, auction);

            conn.commit();  // COMMIT

        } catch (Exception e) {
            conn.rollback();  // ROLLBACK nếu bất kỳ bước nào fail
            throw e;
        }
    } finally {
        auction.getLock().unlock();  // ← luôn unlock dù thành công hay fail
    }
}
